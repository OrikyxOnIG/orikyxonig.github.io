Aimbot with Jitter

function activateAimbot() {
    if (player.isInRangeOfTarget()) {
        let target = player.getTarget();
        let jitterFactor = Math.random() * 0.2; // Adds jitter effect to aiming
        player.aimAt(target.x + jitterFactor, target.y + jitterFactor);
    }
}

Unlock All (Automatically Unlock Everything)

function unlockAll() {
    for (let i = 0; i < weapons.length; i++) {
        player.unlockWeapon(weapons[i]);
    }
    for (let i = 0; i < perks.length; i++) {
        player.unlockPerk(perks[i]);
    }
    player.unlockAllAchievements();
}
